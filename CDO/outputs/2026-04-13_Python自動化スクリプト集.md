## 📜 スクリプト1：daily_report.py

### 目的
毎朝、前日の進捗をまとめて表示

### 必要なもの
- Python 3.10以上（標準ライブラリのみ使用）

### コード
```python
#!/usr/bin/env python3
"""
日次レポート自動生成スクリプト
2026-04-13

使い方：
  1. applications.csv に応募ログを記録
  2. python3 daily_report.py で実行
"""

import csv
from datetime import datetime, timedelta
from collections import defaultdict

CSV_FILE = 'applications.csv'

def load_applications():
    """応募ログCSVを読み込む"""
    applications = []
    try:
        with open(CSV_FILE, 'r', encoding='utf-8') as f:
            reader = csv.DictReader(f)
            for row in reader:
                applications.append(row)
    except FileNotFoundError:
        print(f"❌ {CSV_FILE} が見つかりません。先にCSVを作成してください。")
        return None
    return applications

def analyze_today(applications):
    """今日の実績を集計"""
    today = datetime.now().strftime('%Y-%m-%d')
    today_apps = [a for a in applications if a['date'] == today]
    
    stats = {
        'applications': len(today_apps),
        'replies': len([a for a in today_apps if a['status'] in ['返信', '受注']]),
        'orders': len([a for a in today_apps if a['status'] == '受注']),
        'by_site': defaultdict(int)
    }
    
    for app in today_apps:
        stats['by_site'][app['site']] += 1
    
    return stats

def analyze_week(applications):
    """今週の実績を集計"""
    today = datetime.now()
    week_ago = today - timedelta(days=7)
    
    week_apps = [
        a for a in applications 
        if datetime.strptime(a['date'], '%Y-%m-%d') >= week_ago
    ]
    
    return {
        'applications': len(week_apps),
        'orders': len([a for a in week_apps if a['status'] == '受注']),
        'revenue': sum(int(a.get('price', 0) or 0) for a in week_apps if a['status'] == '受注')
    }

def print_report(today_stats, week_stats):
    """レポートを表示"""
    print("=" * 50)
    print(f"📊 日次レポート（{datetime.now().strftime('%Y-%m-%d')}）")
    print("=" * 50)
    
    print(f"\n【今日の実績】")
    print(f"  応募数: {today_stats['applications']}件")
    print(f"  返信数: {today_stats['replies']}件")
    print(f"  受注数: {today_stats['orders']}件")
    
    if today_stats['by_site']:
        print(f"\n【サイト別応募】")
        for site, count in today_stats['by_site'].items():
            print(f"  {site}: {count}件")
    
    print(f"\n【今週の実績】")
    print(f"  応募数: {week_stats['applications']}件")
    print(f"  受注数: {week_stats['orders']}件")
    print(f"  売上: ¥{week_stats['revenue']:,}")
    
    if week_stats['applications'] > 0:
        win_rate = (week_stats['orders'] / week_stats['applications']) * 100
        print(f"  受注率: {win_rate:.1f}%")
    
    print("=" * 50)

def main():
    applications = load_applications()
    if not applications:
        return
    
    today_stats = analyze_today(applications)
    week_stats = analyze_week(applications)
    print_report(today_stats, week_stats)

if __name__ == '__main__':
    main()
```

## 📜 スクリプト2：rate_calculator.py

### 目的
案件の時給・採算性を計算して、応募すべきか判定

### コード
```python
#!/usr/bin/env python3
"""
時給・採算性計算ツール
2026-04-13
"""

def calculate_hourly_rate():
    print("=" * 50)
    print("💰 時給・採算性計算")
    print("=" * 50)
    
    # 入力
    price = int(input("報酬額（円）: "))
    hours = float(input("想定作業時間（時間）: "))
    platform = input("プラットフォーム（cw/lancers/mama/direct）: ")
    
    # 手数料率
    fees = {
        'cw': 0.20,       # クラウドワークス 20%
        'lancers': 0.165, # ランサーズ 16.5%
        'mama': 0.0,      # ママワークス 0%
        'direct': 0.0     # 直接契約 0%
    }
    fee_rate = fees.get(platform, 0)
    
    # 計算
    fee = price * fee_rate
    net_income = price - fee
    hourly_rate = net_income / hours
    
    # 判定
    print(f"\n【計算結果】")
    print(f"  報酬総額: ¥{price:,}")
    print(f"  手数料: ¥{fee:,.0f}（{fee_rate*100:.1f}%）")
    print(f"  実収入: ¥{net_income:,.0f}")
    print(f"  想定作業時間: {hours}時間")
    print(f"  時給: ¥{hourly_rate:,.0f}/h")
    
    # 採算判定
    print(f"\n【採算判定】")
    if hourly_rate >= 3000:
        print("  ✅ 優良案件：即応募すべき")
    elif hourly_rate >= 2000:
        print("  👍 良い案件：応募推奨")
    elif hourly_rate >= 1000:
        print("  🤔 普通：検討の余地あり")
    elif hourly_rate >= 500:
        print("  ⚠️ 安い：評価作りなら応募可")
    else:
        print("  ❌ 採算割れ：スキップ推奨")
    
    print("=" * 50)

if __name__ == '__main__':
    calculate_hourly_rate()
```

## 📜 スクリプト3：application_tracker.py

### 目的
応募を対話的に記録＆統計表示

### コード
```python
#!/usr/bin/env python3
"""
応募トラッカー
2026-04-13
"""

import csv
import os
from datetime import datetime

CSV_FILE = 'applications.csv'

def init_csv():
    """CSVが無ければ作成"""
    if not os.path.exists(CSV_FILE):
        with open(CSV_FILE, 'w', encoding='utf-8', newline='') as f:
            writer = csv.writer(f)
            writer.writerow(['date', 'site', 'job_name', 'price', 'status'])
        print(f"✅ {CSV_FILE} を作成しました")

def add_application():
    """応募を記録"""
    print("\n📝 新規応募を記録")
    date = input("日付（YYYY-MM-DD、空白で今日）: ") or datetime.now().strftime('%Y-%m-%d')
    site = input("サイト（cw/lancers/mama等）: ")
    job_name = input("案件名: ")
    price = input("報酬額: ") or '0'
    status = input("ステータス（応募/返信/受注/失注）: ") or '応募'
    
    with open(CSV_FILE, 'a', encoding='utf-8', newline='') as f:
        writer = csv.writer(f)
        writer.writerow([date, site, job_name, price, status])
    
    print(f"✅ 記録しました: {job_name}")

def update_status():
    """既存応募のステータス更新"""
    print("\n🔄 ステータス更新")
    job_name = input("更新する案件名（部分一致）: ")
    new_status = input("新しいステータス: ")
    
    # CSV読み込み
    rows = []
    with open(CSV_FILE, 'r', encoding='utf-8') as f:
        reader = csv.DictReader(f)
        for row in reader:
            if job_name in row['job_name']:
                row['status'] = new_status
                print(f"✏️ 更新: {row['job_name']} → {new_status}")
            rows.append(row)
    
    # 書き戻し
    with open(CSV_FILE, 'w', encoding='utf-8', newline='') as f:
        writer = csv.DictWriter(f, fieldnames=['date', 'site', 'job_name', 'price', 'status'])
        writer.writeheader()
        writer.writerows(rows)

def show_stats():
    """統計表示"""
    with open(CSV_FILE, 'r', encoding='utf-8') as f:
        reader = csv.DictReader(f)
        rows = list(reader)
    
    total = len(rows)
    orders = len([r for r in rows if r['status'] == '受注'])
    revenue = sum(int(r.get('price', 0) or 0) for r in rows if r['status'] == '受注')
    
    print("\n" + "=" * 50)
    print("📊 統計")
    print("=" * 50)
    print(f"総応募数: {total}件")
    print(f"受注数: {orders}件")
    if total > 0:
        print(f"受注率: {orders/total*100:.1f}%")
    print(f"総売上: ¥{revenue:,}")
    print("=" * 50)

def main():
    init_csv()
    
    while True:
        print("\n" + "=" * 50)
        print("🎯 応募トラッカー")
        print("=" * 50)
        print("1. 新規応募を記録")
        print("2. ステータス更新")
        print("3. 統計表示")
        print("4. 終了")
        
        choice = input("\n選択: ")
        
        if choice == '1':
            add_application()
        elif choice == '2':
            update_status()
        elif choice == '3':
            show_stats()
        elif choice == '4':
            break
        else:
            print("❌ 無効な選択")
    
    print("👋 お疲れ様でした！")

if __name__ == '__main__':
    main()
```

## 📜 スクリプト5：git backup（シンプル版）

### 目的
毎日自動でGitHubにpush

### コード
```python
#!/usr/bin/env python3
"""
日次Gitバックアップ
2026-04-13
"""

import subprocess
from datetime import datetime

def run_backup():
    date = datetime.now().strftime('%Y-%m-%d %H:%M')
    
    # Gitステータス確認
    result = subprocess.run(
        ['git', 'status', '--porcelain'],
        capture_output=True, text=True
    )
    
    if not result.stdout.strip():
        print("📭 変更なし。スキップします。")
        return
    
    # 変更をコミット
    subprocess.run(['git', 'add', '.'])
    subprocess.run([
        'git', 'commit', '-m', 
        f'backup: 自動バックアップ {date}'
    ])
    
    # プッシュ
    subprocess.run(['git', 'push'])
    
    print(f"✅ バックアップ完了: {date}")

if __name__ == '__main__':
    run_backup()
```
